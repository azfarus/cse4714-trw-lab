\section{Multicolumn and Multirow}
You have to use the \verb|multirow| package to combine rows and columns.

\begin{table}[H]
    \centering
    \begin{tabular}{|c|c|c|}
        \hline
        a & b & c \\ \hline 
        \multicolumn{2}{|R|}{\multirow{2}{*}{a \& b}} & c \\ \cline{3-3}
        \multicolumn{3}{|c|}{}  \\ \hline
        a & b & c \\ \hline
        a & b & c \\ \hline
        a & b & c \\ \hline
        a & b & c \\ \hline

    \end{tabular}
    
\end{table}
