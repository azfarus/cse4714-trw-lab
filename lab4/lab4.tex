\section{Basic Table}

\subsection{Basic Table with centered columns}
\begin{center}
    \begin{tabular}{ || c | c | c ||}
        a & b & c \\
        \hline
        \hline
        a & b & c\\[1cm] % vertical space
        \hline
        a & b & c 
        
    \end{tabular}
\end{center}

\subsection{Other Column Specifiers}
These columnwidths scale automatically.
\begin{center}
    \begin{tabular}{ || l | c | r ||}
        Left Justified & centered & Right Justified \\
        \hline
        \hline
        a & b & c\\[1cm] % vertical space
        \hline
        a & b & c 
        
    \end{tabular}
\end{center}
