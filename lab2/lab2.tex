\section{Inline and Display Math Environments}


\subsection{Inline Math}
This is an inline math: $a^2 + b^2 = c^2 $, wrapped using the \$..\$ symbols. \\
This is an inline math: \(a^2 + b^2 = c^2 \), wrapped using the \verb|\(..\)| symbols. \\
Using \texttt{math} environment:
\begin{math}
e^{i\pi} + 1 = 0
\end{math}

\subsection{Display Math}

\begin{itemize}
    \item Using \verb|\[..]\|:
    \[
         e^{i\pi} + 1 = 0
    \]
    \item Using the \texttt{equation} environment:
        \begin{equation}
            e^{i\pi} + 1 = 0 
        \end{equation} 
        \begin{equation}
            e^{i\theta} = \cos(\theta) + i\sin(\theta)
        \end{equation}


        \item Using the \texttt{equation*} environment:
        \begin{equation*}
            e^{i\pi} + 1 = 0 
        \end{equation*}
        \begin{equation*}
            e^{i\theta} = \cos(\theta) + i\sin(\theta)
        \end{equation*}
    \end{itemize}

    x = y, \quad x \neq y, \quad x > y, \quad x < y, \quad x \ge y, \quad x \le y 
\end{equation}


% Text in math
\begin{equation}
    f(x) = x^2 \quad \text{for all } x \in \mathbb{R}
\end{equation}
