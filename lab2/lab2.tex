\subsection{Split and Aligned}

\paragraph{Split}
\begin{equation}
    f(x) = \left\{ \begin{split}
        a + b = 10&  \quad \text{if} \ x > 10\\
        c + d = \  9& \quad  \text{if} \ x \leq 10
    \end{split} \right   
\end{equation}

\paragraph{Align}

\begin{align} % alignment specifiers using the ampersand sign
a &= b + c \\
  &= d + e
\end{align} 

\paragraph{Gather}

\begin{gather} % no alignment specification
    \sin{x} + \cos{x} = 1.414 \\
    \sin{x} + \cos{x} = 1.4 
\end{gather}
