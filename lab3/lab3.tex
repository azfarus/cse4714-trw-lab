\section{Including Basic Images}
\subsection{With Extension}
\lipsum[1]
\begin{figure}[H]
    \centering
    \caption{Fig 1}
    \includegraphics[width=0.2\linewidth]{ImageA.png}
    \label{fig:1}
\end{figure}

\subsection{Without Extension} % demo for widths and scale
\begin{figure}[H]
    \centering
    \includegraphics[scale=1]{ImageA}
    \caption{Fig: 2}
    \label{fig:2}
\end{figure}

\subsection{Rotation}
\lipsum[1]
\begin{figure}[H]
    \centering
    \rotatebox[origin=tr]{35}{\includegraphics[width=0.2\linewidth]{ImageA.png}}
    \caption{Fig: 3}
    \label{fig:3}
\end{figure}
I will now reference to \autoref{fig:1}.
